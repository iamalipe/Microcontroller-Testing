String buildFileListJSON()
{
  String json = "{\"files\":[";
  File root = SD.open("/");
  bool first = true;
  while (true)
  {
    File f = root.openNextFile();
    if (!f)
      break;
    if (!f.isDirectory())
    {
      if (!first)
        json += ",";
      json += "{\"name\":\"/" + String(f.name()) + "\",\"size\":" + f.size() + "}";
      first = false;
    }
    f.close();
  }
  root.close();
  uint64_t total = SD.totalBytes();
  uint64_t used = SD.usedBytes();
  json += "],\"total\":" + String((uint32_t)total) + ",\"used\":" + String((uint32_t)used) + "}";
  return json;
}

void setup()
{
  Serial.begin(115200);
  delay(2000);
  // SD init
  SPI.begin(SD_SCK, SD_MISO, SD_MOSI, SD_CS);
  if (!SD.begin(SD_CS))
  {
    Serial.println("SD init failed!");
    while (true)
      delay(1000);
  }
  Serial.println("SD OK  type:" + String(SD.cardType()));

  // WiFi
  WiFi.mode(WIFI_STA);
  WiFi.begin(SSID, PASSWORD);
  Serial.print("Connecting");
  while (WiFi.status() != WL_CONNECTED)
  {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nIP: " + WiFi.localIP().toString());

  // mDNS
  if (MDNS.begin(HOSTNAME))
    Serial.println("mDNS: http://" + String(HOSTNAME) + ".local");

  // ── Routes ──────────────────────────────────────────────

  server.on("/", HTTP_GET, [](AsyncWebServerRequest *req)
            { req->send(200, "text/html", INDEX_HTML); });

  server.on("/list", HTTP_GET, [](AsyncWebServerRequest *req)
            { req->send(200, "application/json", buildFileListJSON()); });

  server.on("/download", HTTP_GET, [](AsyncWebServerRequest *req)
            {
    if (!req->hasParam("path")) { req->send(400, "text/plain", "Missing path"); return; }
    String path = req->getParam("path")->value();
    if (!SD.exists(path)) { req->send(404, "text/plain", "Not found"); return; }
    req->send(SD, path, "application/octet-stream", true); });

  server.on("/delete", HTTP_DELETE, [](AsyncWebServerRequest *req)
            {
    if (!req->hasParam("path")) { req->send(400, "text/plain", "Missing path"); return; }
    String path = req->getParam("path")->value();
    if (!SD.exists(path)) { req->send(404, "text/plain", "Not found"); return; }
    SD.remove(path);
    req->send(200, "text/plain", "Deleted: " + path); });

  // Upload handler
  server.on("/upload", HTTP_POST, [](AsyncWebServerRequest *req)
            { req->send(200, "text/plain", "OK"); }, [](AsyncWebServerRequest *req, String filename, size_t index, uint8_t *data, size_t len, bool final)
            {
      if (!index) {
        String path = "/" + filename;
        if (SD.exists(path)) SD.remove(path);
        uploadFile = SD.open(path, FILE_WRITE);
        Serial.println("Upload start: " + path);
      }
      if (uploadFile) uploadFile.write(data, len);
      if (final) {
        if (uploadFile) uploadFile.close();
        Serial.printf("Upload done: %s (%u bytes)\n", filename.c_str(), index + len);
      } });

  server.onNotFound([](AsyncWebServerRequest *req)
                    { req->send(404, "text/plain", "Not found"); });

  server.begin();
  Serial.println("Server started");
}
